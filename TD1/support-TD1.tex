\section{Les types composés/structurés}

%\subsection{En algorithmique}

\frame
{
  \frametitle{Rappel sur les types simples (p21-24)}
Les types \textbf{simples} de données~:\\
\begin{tabular}{p{5cm}|l}
type {\entier} & \texttt{int}\\
type {\reel} & \texttt{float}\\
type {\car} & \texttt{char}\\
type {\bool} & \texttt{int}
\end{tabular}
\\
Il est temps de découvrir les \textbf{types composés}, permettant de manipuler des objets plus complexes.
}


\frame[allowframebreaks]{
  \frametitle{Pourquoi des types composés ? (p21)}
  
  \begin{itemize}
  	\item Pour manipuler des collections d'\textbf{objets ayant le même type} $\rightarrow$ \textbf{tableaux} ! \\ Par exemple : un vecteur d'entiers
  	\item Pour regrouper des \textbf{objets de types différents dans une même structure} $\rightarrow$ \textbf{enregistrements} ! \\ Par exemple : un objet "Article" avec :
	\begin{itemize}
		\item sa référence (un entier)
		\item son libellé (une chaîne de caractères)
		\item son prix (un réel).
	\end{itemize}
	
	\framebreak
	
	\item Convention : les types ainsi construits auront un nom commen\c cant par une majuscule systématiquement précédé par {\tt {\ty}}
	\item Définition des types composés \textbf{dans la partie "Spécifications"} de l'algorithme
  \end{itemize}
}

\frame[allowframebreaks]{
  \frametitle{Les types tableaux (p21)}
  
  \begin{itemize}
  	\item Pour manipuler des données :
		\begin{itemize}
		\item \textbf{de même type} ; 
		\item devant être traitées de manière identique ;
		\item \textbf{de nombre maximum connu !} ; 
		\end{itemize}
	\item Un tableau est défini par :
		\begin{itemize}
		\item son nom ; 
		\item \textbf{le type de ses éléments} ;
		\item la \textbf{taille maximale} de chaque dimension (facultatif en algo). 
		\end{itemize}
  \end{itemize}
  
  \framebreak
  
  \begin{itemize}
  	\item Mots réservés~: \vecteur, \matrice, et plus généralement \tableau pour les dimensions supérieures.
	\item \textbf{Vecteur} :
		\begin{itemize}
		\item ensemble d'éléments \textbf{de même type} ; 
		\item une composante $=$ un élément identifié par un indice (entier) 
		\item $i^{\grave eme}$ composante d'un vecteur  {\tt v} : ${\tt v[i]}$ ou ${\tt v_i}$.
		\end{itemize}
	\item \textbf{Matrice} :
		\begin{itemize}
		\item ensemble d'éléments \textbf{de même type} construits selon $2$ directions ; 
		\item un élément : repéré par 2 coordonnées ;
		\item élément de la $i^{\grave eme}$ ligne et de la $j^{\grave eme}$
colonne d'une matrice {\tt M} : ${\tt M_{i,j}}$ ou ${\tt M[i][j]}$.
		\end{itemize}
  \end{itemize}
  
    \framebreak
    
    \norm{/ déclarations de types tableaux /}
    \norm{\types}
    \decal{\ty VectEnt~: \vecteur d'\entiers}
    \decal{\ty MatReels~: \matrice de \reels}
    \decal{\ty MonTableau~: \tableau d'\entiers de dimension 4}
}

\frame[allowframebreaks]{
  \frametitle{Les types enregistrements (p22)}
  
  \begin{itemize}
  	\item \textbf{Pour regrouper des données de types différents.}
	\item Pour manipuler un ensemble d'"Articles" (libellé/prix/référence) :
		\begin{itemize}
		\item manipuler trois tableaux différents pour l'ensemble des articles ;
		\item ou, plus malin : \textbf{utiliser un enregistrement avec trois champs}
		\end{itemize}
  \end{itemize}
  
  \framebreak
  
\norm{// Déclarations de types }

\decal{\ty Chaine~: \vecteur de \cars}
\decal{\ty Article~: \enreg}
\decala{\entier~:  reference,}
\decala{\commentz{\ty Chaine~: libelle,}{type précédemment \\défini comme vecteur \\de caractères}}
\decala{\reel~: prix}

\decal{\ty VectArticles~: \vecteur de \ty Article}\label{typeVectArticles}
%\\

  \framebreak
  
  \textbf{Déclaration générale d'un type enregistrement} : \\
  \norm{\type}
\decal{nomduType~: \enreg}
\decala{\commentz{type champ1,}{de type simple ou composé \\précédemment défini}}
\decala{\commentz{type champ2,}{de type simple ou composé \\précédemment défini}}

\decal{\finenreg}

  \framebreak
  
  \textbf{Manipulation d'une variable de type enregistrement} :
  \\
  Soit \texttt{monArticle} une variable de type \texttt{\ty Article}
  \begin{itemize}
  	\item \texttt{monArticle.libelle}, de type \texttt{\ty Chaine}, permet d'accéder au libellé de l'article ;
  	\item \texttt{monArticle.reference}, de type \entier, permet d'accéder à la référence de l'article ;
	\item \texttt{monArticle.prix}, de type \reel, permet d'accéder au prix de l'article.
    \end{itemize}
}

\frame{
  \frametitle{Affectation de variables de types composés}
  Soient {\tt s1} et {\tt s2} deux variables de types composés~:
  \begin{itemize}
  	\item si {\tt s1} et {\tt s2} sont \textbf{deux tableaux de même type}, {\tt s1$\leftarrow$ s2} signifie que chaque élément de {\tt s1} prend pour valeur celle de l'élément de {\tt s2} de même indice,
	\item si {\tt s1} et {\tt s2} sont \textbf{deux enregistrements de même type}, {\tt s1$\leftarrow$ s2} signifie que chaque champ de {\tt s1} prend pour valeur celle du champ correspondant de {\tt s2}.
  \end{itemize}
}

\frame{
  \frametitle{Exemple et remarques}
  \begin{itemize}
  	\item Application sur le calcul des prix minimum et maximum d'une liste d'articles : p35-37
	\item Lors d'une lecture/écriture d'une variable de type composé, ne pas développer élément par élément.
	\item \norm{listeArticles $\leftarrow$ \lire{}} suffit. 
	\item Les fonctions peuvent bien sûr renvoyer des types composés (mais \alert{attention à la traduction en C !})
  \end{itemize}
}

\section{Traitements de files}

\subsection{Introduction}

\frame[allowframebreaks]{
  \frametitle{Qu'est-ce qu'une file? (p38)}
  
  \begin{itemize}
  	\item Quelques exemples de files :
		\begin{itemize}
			\item la file d'attente à la boulangerie ;
			\item une file de données numériques (notes d'un étudiant) ;
			\item \dots
		\end{itemize}
	\item Le nombre d'éléments de la file n'est pas forcément connu \textit{a priori}.
  \framebreak
  	
	\item Un ensemble d'éléments a une structure de \textbf{file} non vide si on peut définir :		
  		\begin{itemize}
			\item   le premier élément de la file,
			\item   un procédé permettant d'obtenir l'élément suivant de la file,
			\item   un procédé permettant de détecter la fin de la file.
		\end{itemize}
	\item On veut appliquer à tous les éléments de la file (sauf éventuellement le dernier) un traitement.
		
	\item \textbf{Idée fondamentale} pour le traitement de file : structures "\tantque{\ldots} / \fintantque" et "\pour{\ldots}{\ldots}/\finpour"
  \end{itemize}
}

\subsection{Traitement de files de longueur inconnue}

\frame{
  \frametitle{Conditions (p38)}
  
  \begin{itemize}
  	\item Longueur de la file inconnue a priori
	\item Elément final - \textbf{la butée} - supposé reconnaissable
	\item \textbf{Idée fondamentale} : structure "\tantque{\ldots} / \fintantque"
	\item Ne pas oublier les cas aux limites : file vide
  \framebreak
  \end{itemize}
}
  	
\frame{
  \frametitle{Schéma d'algorithme (p39)}
\debut
\inst{{\em //initialisations des données}}
\inst{{\em //obtenir le premier élément de la file}}
\inst{\ldots\ldots\ldots}
\inst{{\em //itérations sur les éléments de la file à traiter}}
\inst{\tantque{<<élément non final>>}}
\instr{{\em //traitement de l'élément courant}}
\instr{{\em //obtenir l'élément suivant}}
\instr{\ldots\ldots\ldots}
\inst{\fintantque}
\inst{{\em //traitement éventuel de l'élément final}}
\inst{\ldots\ldots\ldots}
\inst{{\em //traitement éventuel lié à la fin de file}}
\inst{\ldots\ldots\ldots}
\fin
}

\subsection{Traitement de files de longueur connue}

\frame{
  \frametitle{Conditions (p39)}
  
  \begin{itemize}
  	\item Longueur de la file connue
	\item \textit{Chaque élément} : l'obtenir et le traiter
	\item \textbf{Idée fondamentale} : structure "\pour{\ldots}{\ldots}/\finpour"
	\item Ne pas oublier les cas aux limites : file vide
  \end{itemize}
